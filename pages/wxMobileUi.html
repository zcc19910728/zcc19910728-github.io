对话框
 
                
                <p><a id="toastBtn" class="button button-red-lg">普通提示框</a></p>
                <p><a id="alertBtn" class="button button-red-lg">alert提示框</a></p>
                <p><a id="confirmBtn" class="button button-red-lg">confirm提示框</a></p>
                <div id="tips" class="cor-red pl-20"></div>
                <script type="text/javascript">
                    $(function(){
                        $('#toastBtn').click(function(){
                            Yt.toast('普通提示框', 3000, function(){
                                $('#tips').text('普通提示框');
                            });
                        });

                        $('#alertBtn').click(function(){
                            Yt.alert('我是alert提示框', function(){
                                $('#tips').text('alert提示框');
                            });
                        });

                        $('#confirmBtn').click(function(){
                            Yt.confirm('我是alert提示框', function(){
                                $('#tips').text('confirm提示框');
                                Yt.dialogClose("ytDialog");
                            });
                        });
                    });
                </script>
